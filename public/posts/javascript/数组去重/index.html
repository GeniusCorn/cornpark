filter + Object

function unique(array) {
  let container = {};
  return array.filter((item) =>
    container.hasOwnProperty(item) ? false : (container[item] = true),
  );
}

filter + indexOf

function unique(array) {
  return array.filter((item, index) => array.indexOf(item) === index);
}

Set

function unique(array) {
  return [...new Set(array)];
}
